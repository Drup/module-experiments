  \Oenrich{\E}{t = \tau'}{\Stype{t}{\tau};\iS}

  &\text{when } \submod{\E}{\iM'}{\iM}{\_}\\

  &\to \Smodulety{\ix}{\Oenrich{\E}{\ip : \iM'}{\iM}};\iS\\
